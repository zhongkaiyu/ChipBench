I would like you to implement a module named TopModule with the following
interface. All input and output ports are one bit unless otherwise
specified.

 - input  clk
 - input  rst_n
 - input  wave_choice (2 bits)
 - output wave (5 bits)

The module should implement a signal generator that outputs different
waveforms based on the wave_choice input signal. The generator should
produce three different types of waveforms:

When wave_choice = 0: Generate a square wave signal
When wave_choice = 1: Generate a sawtooth wave signal  
When wave_choice = 2: Generate a triangular wave signal

The waveform generation should work as follows:
- The wave output should be a 5-bit signal that cycles through different
  values to create the desired waveform patterns
- Each waveform should complete one full cycle and then repeat
- The square wave should alternate between high and low values
- The sawtooth wave should increment from 0 to maximum value and then reset
- The triangular wave should increment from 0 to maximum, then decrement back to 0


Based on the problem description above, the code below in TopModule has bug, please fix that:
```
module TopModule(
	input clk,
	input rst_n,
	input [1:0] wave_choice,
	output reg [4:0]wave
	);

    reg [4:0] cnt;
    reg flag;
    
  	// Square wave mode, counter control
    always@(posedge clk or negedge rst_n) begin
        if(~rst_n)
            cnt <= 0;
        else
            cnt <= wave_choice!=0 ? 0:
                   cnt        ==20? 0:
                   cnt + 1;
    end
    
  	// Triangle wave mode, flag bit control
    always@(posedge clk or negedge rst_n) begin
        if(~rst_n)
            flag <= 0;
        else
            flag <= wave_choice!=2 ? 0:
                    wave       ==1 ? 1:
                    wave       ==20? 0:
                    flag;
    end
    
  
  	// Update wave signal
    always@(posedge clk or negedge rst_n) begin
        if(~rst_n) 
            wave <= 0;
        else 
            case(wave_choice)
                0      : wave <= cnt == 9? 20    : 
                                 cnt ==20? 0     :
                                 wave;
                1      : wave <= wave==20? 0     : wave+1;
                2      : wave <= flag==0 ? wave-1: wave+1;
                default: wave <= 0;
            endcase
    end
endmodule
```
