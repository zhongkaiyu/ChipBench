- Output signal: flag (reg type)
- Reset: rst is active low
- Clock: clk for synchronous operation

  S3               S4    S3

	input wire clk  ,
	input wire rst  ,
	input wire data ,
	output reg flag
);

    reg [2:0]	current_state;
    reg [2:0]	next_state;
	
    always@(posedge clk or negedge rst)begin

            current_state <= S0;

            current_state <= next_state;
